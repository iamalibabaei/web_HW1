def valid_paren(input_str):
  stack = []
  for paren in input_str:
    if paren == '(':
      stack.append(paren)
    else:
      if not stack:
        return False
      else:
        top = stack[-1]
        if paren == ')' and top == '(':
          stack.pop()
  if not stack:
    return True
  else:
    return False
